   41 data BuilderSt tag v x t = BuilderSt
   42     { signalBusWidth :: Int
   43     , availPorts :: [SignalTag]
   44     , puProtos :: [(tag, PU v x t)]
   45     , netEnv :: UnitEnv (BusNetwork tag v x t)
   46     }
   47 
   48 -- |Define microarchitecture with BusNetwork
   49 defineNetwork bnName ioSync builder =
   50     let defEnv = def
   51         st0 =
   52             BuilderSt
   53                 { signalBusWidth = 0
   54                 , availPorts = map (SignalTag . controlSignalLiteral) [0 :: Int ..]
   55                 , puProtos = []
   56                 , netEnv = defEnv
   57                 }
   58         BuilderSt{signalBusWidth, puProtos} = flip execState st0 $ void builder
   59         netIOPorts =
   60             BusNetworkIO
   61                 { extInputs = L.nub $ concatMap (puInputPorts . snd) puProtos
   62                 , extOutputs = L.nub $ concatMap (puOutputPorts . snd) puProtos
   63                 , extInOuts = L.nub $ concatMap (puInOutPorts . snd) puProtos
   64                 }
   65      in BusNetwork
   66             { bnName
   67             , bnRemains = []
   68             , bnBinded = M.empty
   69             , bnProcess = def
   70             , bnPus = M.fromList puProtos
   71             , bnSignalBusWidth = signalBusWidth
   72             , ioSync
   73             , bnEnv = defEnv{ioPorts = Just netIOPorts}
   74             }
   75 
   76 -- |Add PU with the default initial state. Type specify by IOPorts.
   77 add tag ioport = addCustom tag def ioport
   78 
   79 -- |Add PU with the custom initial state. Type specify by IOPorts.
   80 addCustom tag pu ioports = do
   81     st@BuilderSt{signalBusWidth, availPorts, puProtos, netEnv} <- get
   82     let ports = takePortTags availPorts pu
   83         pu' =
   84             PU
   85                 pu
   86                 def
   87                 netEnv
   88                     { ctrlPorts = Just ports
   89                     , ioPorts = Just ioports
   90                     , valueIn = Just ("data_bus", "attr_bus")
   91                     , valueOut = Just (toText tag <> "_data_out", toText tag <> "_attr_out")
   92                     }
   93         usedPorts = usedPortTags ports
   94     put
   95         st
   96             { signalBusWidth = signalBusWidth + length usedPorts
   97             , availPorts = drop (length usedPorts) availPorts
   98             , puProtos = (tag, pu') : puProtos
   99             }

  101 data MicroarchitectureDesc tag = MicroarchitectureDesc
  102     { networks :: [NetworkDesc tag]
  103     , ioSyncMode :: IOSynchronization
  104     }
  105     deriving (Generic)
  106 
  107 instance (ToJSON tag) => ToJSON (MicroarchitectureDesc tag)
  108 
  109 data NetworkDesc tag = NetworkDesc
  110     { networkTag :: tag
  111     , valueType :: T.Text
  112     , units :: [UnitDesc tag]
  113     }
  114     deriving (Generic)
  115 
  116 instance (ToJSON tag) => ToJSON (NetworkDesc tag)
  117 
  118 data UnitDesc tag = UnitDesc
  119     { unitTag :: tag
  120     , unitType :: T.Text
  121     }
  122     deriving (Generic)
  123 
  124 instance (ToJSON tag) => ToJSON (UnitDesc tag)
  125 
  126 microarchitectureDesc :: forall tag v x t. (Typeable x) => BusNetwork tag v x t -> MicroarchitectureDesc tag
  127 microarchitectureDesc BusNetwork{bnName, bnPus, ioSync} =
  128     MicroarchitectureDesc
  129         { networks =
  130             [ NetworkDesc
  131                 { networkTag = bnName
  132                 , valueType = showText $ typeRep (Proxy :: Proxy x)
  133                 , units =
  134                     map
  135                         ( \(tag, PU{unit}) ->
  136                             UnitDesc
  137                                 { unitTag = tag
  138                                 , unitType = T.pack $ takeWhile (' ' /=) $ show $ typeOf unit
  139                                 }
  140                         )
  141                         $ M.assocs bnPus
  142                 }
  143             ]
  144         , ioSyncMode = ioSync
  145         }
