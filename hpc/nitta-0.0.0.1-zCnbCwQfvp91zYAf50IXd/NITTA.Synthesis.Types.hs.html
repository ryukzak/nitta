   16 Synthesis can be represented as a graph (tree), where each 'Node' describes the
   17 target system 'ModelState' and each 'Edge' synthesis decision.
   18 
   19 A synthesis graph is very large and calculating and storing it in memory is very
   20 bad idea. Also, working with synthesis graph usually making from the specific
   21 node, not from the root. As a result, synthesis graph design as a explicit lazy
   22 mutable structure implemented by 'TVar'.
   23 
   24 From this point of view, the synthesis process is a finding of the best tree
   25 leaf (lowest process duration for finished synthesis), and the best synthesis
   26 method - a method which directly walks over the tree to the best leaf without
   27 wrong steps.
   28 -}

   61 -- |Default synthesis tree type.
   62 type DefTree tag v x t =
   63     Tree (TargetSystem (BusNetwork tag v x t) tag v x t) tag v x t
   64 
   65 {- |The synthesis method is a function, which manipulates a synthesis tree. It
   66 receives a node and explores it deeply by IO.
   67 -}
   68 type SynthesisMethod tag v x t = DefTree tag v x t -> IO (DefTree tag v x t)

   70 {- |Synthesis node ID. ID is a relative path, encoded as a sequence of an option
   71 index.
   72 -}
   73 newtype SID = SID [Int]
   74 
   75 -- |SID separator for @Show SID@ and @Read SID@.
   76 sidSep = '-'
   77 
   78 instance Show SID where
   79     show (SID []) = [sidSep]
   80     show (SID is) = show' is
   81         where
   82             show' [] = ""
   83             show' (x : xs) = sidSep : show x ++ show' xs
   84 
   85 instance Read SID where
   86     readsPrec _ [x] | x == sidSep = [(SID [], "")]
   87     readsPrec d (x : xs)
   88         | x == sidSep
   89           , let is = map (readsPrec d) $ splitOn [sidSep] xs
   90           , not $ any null is =
   91             [(SID $ map fst $ concat is, "")]
   92     readsPrec _ _ = []
   93 
   94 instance Default SID where
   95     def = SID []
   96 
   97 instance Semigroup SID where
   98     (SID a) <> (SID b) = SID (a <> b)
   99 
  100 instance Monoid SID where
  101     mempty = SID []
  102     mappend = (<>)
  103 
  104 instance ToJSON SID where
  105     toJSON sid = toJSON $ show sid
  106 
  107 instance FromHttpApiData SID where
  108     parseUrlPiece = Right . read . T.unpack

  110 -- |Synthesis tree
  111 data Tree m tag v x t = Tree
  112     { sID :: SID
  113     , sState :: SynthesisState m tag v x t
  114     , sDecision :: SynthesisDecision (SynthesisState m tag v x t) m
  115     , -- |lazy mutable field with different synthesis options and sub nodes
  116       sSubForestVar :: TMVar [Tree m tag v x t]
  117     }
  118 
  119 targetUnit = mUnit . sTarget . sState
  120 targetDFG = mDataFlowGraph . sTarget . sState
  121 
  122 data SynthesisDecision ctx m where
  123     Root :: SynthesisDecision ctx m
  124     SynthesisDecision ::
  125         (Typeable p, SynthesisDecisionCls ctx m o d p, Show d, ToJSON p, Viewable d DecisionView) =>
  126         {option :: o, decision :: d, metrics :: p, score :: Float} ->
  127         SynthesisDecision ctx m
  128 
  129 class SynthesisDecisionCls ctx m o d p | ctx o -> m d p where
  130     decisions :: ctx -> o -> [(d, m)]
  131     parameters :: ctx -> o -> d -> p
  132     estimate :: ctx -> o -> d -> p -> Float
  133 
  134 data SynthesisState m tag v x t = SynthesisState
  135     { sParent :: Maybe (Tree m tag v x t)
  136     , sTarget :: m
  137     , -- |bind options cache
  138       sBindOptions :: [Bind tag v x]
  139     , sResolveDeadlockOptions :: [ResolveDeadlock v x]
  140     , sOptimizeAccumOptions :: [OptimizeAccum v x]
  141     , sConstantFoldingOptions :: [ConstantFolding v x]
  142     , sBreakLoopOptions :: [BreakLoop v x]
  143     , -- |dataflow options cache
  144       sDataflowOptions :: [DataflowSt tag v (TimeConstraint t)]
  145     , -- |a map from functions to possible processor unit tags
  146       bindingAlternative :: M.Map (F v x) [tag]
  147     , -- |a function set, which binding may cause dead lock
  148       possibleDeadlockBinds :: S.Set (F v x)
  149     , -- |if algorithm will be represented as a graph, where nodes -
  150       -- variables of not binded functions, edges - casuality, wave is a
  151       -- minimal number of a step from an initial node to selected
  152       bindWaves :: M.Map v Int
  153     , -- |number of dataflow options
  154       numberOfDataflowOptions :: Int
  155     , -- |a variable set, which can be transferred on the current
  156       -- synthesis step
  157       transferableVars :: S.Set v
  158     }
