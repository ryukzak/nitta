   74 {- | Loop -- function for transfer data between computational cycles.
   75 Let see the simple example with the following implementation of the
   76 Fibonacci algorithm.
   77 
   78 Data flow graph:
   79 
   80 @
   81     +---------------------------------+
   82     |                                 |
   83     v                                 |
   84 +------+                          b2  |
   85 | Loop |      b1_1  +-----+    +------+
   86 +------+----+------>|     |    |
   87             | a1    | sum +----+
   88 +------+----------->|     |
   89 | Loop |    |       +-----+      b1_2
   90 +------+    +-------------------------+
   91     ^                                 |
   92     |                                 |
   93     +---------------------------------+
   94 @
   95 
   96 Lua source code:
   97 
   98 @
   99 function fib(a1, b1)
  100     b2 = a1 + b1
  101     fib(b1, b2)
  102 end
  103 fib(0, 1)
  104 @
  105 
  106 Data flow defines computation for a single computational cycle. But
  107 a controller should repeat the algorithm infinite times, and
  108 usually, it is required to transfer data between cycles. `Loop`
  109 allows doing that. At first cycle, `Loop` function produces an
  110 initial value (`X x`), after that on each cycle `Loop` produces a
  111 variable value from the previous cycle, and consumes a new value at
  112 the end of the cycle.
  113 
  114 Computational process:
  115 
  116 @
  117          ][                 Cycle 1                 ][                Cycle 2                  ]
  118          ][                                         ][                                         ]
  119 initial  ][ ---+                          b2   +--- ][ ---+                          b2   +--- ]
  120  value   ][ op |      b1_1  +-----+    +------>| Lo ][ op |      b1_1  +-----+    +------>| Lo ]
  121  is a    ][ ---+----+------>|     |    |       +--- ][ ---+----+------>|     |    |       +--- ]
  122 part of  ][         |       | sum +----+            ][         |       | sum +----+            ]
  123 software ][ ---+----------->|     |            +--- ][ ---+----------->|     |            +--- ]
  124          ][ op |    |       +-----+     b1_2   | Lo ][ op |    |       +-----+      b1_2  | Lo ]
  125          ][ ---+    +------------------------->+--- ][ ---+    +------------------------->+--- ]
  126          ][                                         ][                                         ]
  127 @
  128 
  129 Similation data:
  130 
  131 +--------------+----+----+----+
  132 | Cycle number | a1 | b1 | b2 |
  133 +==============+====+====+====+
  134 | 1            | 0  | 1  | 1  |
  135 +--------------+----+----+----+
  136 | 2            | 1  | 1  | 2  |
  137 +--------------+----+----+----+
  138 | 3            | 1  | 2  | 3  |
  139 +--------------+----+----+----+
  140 | 4            | 2  | 3  | 5  |
  141 +--------------+----+----+----+
  142 
  143 In practice, Loop function supported by Fram processor unit in the
  144 following way: Loop function should be prepared before execution by
  145 automatical refactor @BreakLoop@, which replace Loop by @LoopEnd@
  146 and @LoopBegin@.
  147 -}
  148 data Loop v x = Loop (X x) (O v) (I v) deriving (Typeable, Eq)
  149 
  150 instance (Var v, Show x) => Show (Loop v x) where show = label
  151 instance (Var v, Show x) => Label (Loop v x) where
  152     label (Loop (X x) os i) =
  153         "loop(" <> show x <> ", " <> show i <> ") = " <> show os
  154 loop :: (Var v, Val x) => x -> v -> [v] -> F v x
  155 loop x a bs = packF $ Loop (X x) (O $ S.fromList bs) $ I a
  156 isLoop f
  157     | Just Loop{} <- castF f = True
  158     | otherwise = False
  159 
  160 instance Function (Loop v x) v where
  161     isInternalLockPossible _ = True
  162     inputs (Loop _ _a b) = variables b
  163     outputs (Loop _ a _b) = variables a
  164 instance Var v => Patch (Loop v x) (v, v) where
  165     patch diff (Loop x a b) = Loop x (patch diff a) (patch diff b)
  166 instance Var v => Locks (Loop v x) v where
  167     locks (Loop _ (O as) (I b)) = [Lock{locked = b, lockBy = a} | a <- S.elems as]
  168 instance Var v => FunctionSimulation (Loop v x) v x where
  169     simulate CycleCntx{cycleCntx} (Loop (X x) (O vs) (I _)) =
  170         case oneOf vs `HM.lookup` cycleCntx of
  171             -- if output variables are defined - nothing to do (values thrown on upper level)
  172             Just _ -> []
  173             -- if output variables are not defined - set initial value
  174             Nothing -> [(v, x) | v <- S.elems vs]

  291 data Division v x = Division
  292     { denom, numer :: I v
  293     , quotient, remain :: O v
  294     }
  295     deriving (Typeable, Eq)
  296 instance Label (Division v x) where label Division{} = "/"
  297 instance Var v => Show (Division v x) where
  298     show Division{denom, numer, quotient, remain} =
  299         let q = show numer <> " / " <> show denom <> " = " <> show quotient
  300             r = show numer <> " mod " <> show denom <> " = " <> show remain
  301          in q <> "; " <> r
  302 division :: (Var v, Val x) => v -> v -> [v] -> [v] -> F v x
  303 division d n q r =
  304     packF $
  305         Division
  306             { denom = I d
  307             , numer = I n
  308             , quotient = O $ S.fromList q
  309             , remain = O $ S.fromList r
  310             }
  311 
  312 instance Var v => Function (Division v x) v where
  313     inputs Division{denom, numer} = variables denom `S.union` variables numer
  314     outputs Division{quotient, remain} = variables quotient `S.union` variables remain
  315 instance Var v => Patch (Division v x) (v, v) where
  316     patch diff (Division a b c d) = Division (patch diff a) (patch diff b) (patch diff c) (patch diff d)
  317 instance Var v => Locks (Division v x) v where
  318     locks = inputsLockOutputs
  319 instance (Var v, Integral x) => FunctionSimulation (Division v x) v x where
  320     simulate cntx Division{denom = I d, numer = I n, quotient = O qs, remain = O rs} =
  321         let dx = cntx `getCntx` d
  322             nx = cntx `getCntx` n
  323             (qx, rx) = dx `quotRem` nx
  324          in [(v, qx) | v <- S.elems qs] ++ [(v, rx) | v <- S.elems rs]

  451 data CmpOp = CmpEq | CmpLt | CmpLte | CmpGt | CmpGte
  452     deriving (Typeable, Eq, Show, Data, Generic)
  453 
  454 data Compare v x = Compare CmpOp (I v) (I v) (O v) deriving (Typeable, Eq)
  455 instance Label (Compare v x) where
  456     label (Compare op _ _ _) = show op
  457 instance Var v => Patch (Compare v x) (v, v) where
  458     patch diff (Compare op a b c) = Compare op (patch diff a) (patch diff b) (patch diff c)
  459 
  460 instance Var v => Show (Compare v x) where
  461     show (Compare op a b o) = show a <> " " <> show op <> " " <> show b <> " = " <> show o
  462 
  463 instance Var v => Function (Compare v x) v where
  464     inputs (Compare _ a b _) = variables a `S.union` variables b
  465     outputs (Compare _ _ _ o) = variables o
  466 instance (Var v, Val x) => FunctionSimulation (Compare v x) v x where
  467     simulate cntx (Compare op (I a) (I b) (O o)) =
  468         let
  469             x1 = getCntx cntx a
  470             x2 = getCntx cntx b
  471             y = if op2func op x1 x2 then 1 else 0
  472          in
  473             [(v, y) | v <- S.elems o]
  474         where
  475             op2func CmpEq = (==)
  476             op2func CmpLt = (<)
  477             op2func CmpLte = (<=)
  478             op2func CmpGt = (>)
  479             op2func CmpGte = (>=)
  480 instance Var v => Locks (Compare v x) v where
  481     locks = inputsLockOutputs
  482 
  483 cmp :: (Var v, Val x) => CmpOp -> v -> v -> [v] -> F v x
  484 cmp op a b c = packF $ Compare op (I a) (I b) $ O $ S.fromList c
