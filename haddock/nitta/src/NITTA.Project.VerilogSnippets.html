module NITTA.Project.VerilogSnippets (
    snippetClkGen,
    snippetDumpFile,
) where

import Data.String.Interpolate
import Data.Text qualified as T

snippetClkGen :: T.Text
snippetClkGen :: Text
snippetClkGen =
    [__i|
        initial begin
            clk = 1'b0;
            rst = 1'b1;
            repeat(4) \#1 clk = ~clk;
            rst = 1'b0;
            forever \#1 clk = ~clk;
        end
    |]

snippetDumpFile :: T.Text -> T.Text
snippetDumpFile :: Text -> Text
snippetDumpFile Text
mn =
    [__i|
        initial begin
            $dumpfile("#{ mn }_tb.vcd");
            $dumpvars(0, #{ mn }_tb);
        end
    |]
