data BackendCtx tag v x t = BackendCtx
    { forall tag v x t. BackendCtx tag v x t -> DefTree tag v x t
root :: DefTree tag v x t
    -- ^ root synthesis node
    , forall tag v x t. BackendCtx tag v x t -> [(v, [x])]
receivedValues :: [(v, [x])]
    -- ^ lists of received by IO values
    , forall tag v x t. BackendCtx tag v x t -> String
outputPath :: String
    , forall tag v x t. BackendCtx tag v x t -> IO MlBackendServer
mlBackendGetter :: IO MlBackendServer
    , forall tag v x t. BackendCtx tag v x t -> [Text]
nodeScores :: [T.Text]
    {- ^ list of node scores name. Used for ML backend mostly, but can be used
    for any "hardcoded" scoring function.
    -}
    }

instance Default (BackendCtx tag v x t) where
    def :: BackendCtx tag v x t
def =
        BackendCtx
            { root :: DefTree tag v x t

            , receivedValues :: [(v, [x])]

            , outputPath :: String

def
            , mlBackendGetter :: IO MlBackendServer

            , nodeScores :: [Text]
